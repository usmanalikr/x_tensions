def rcall(*args)
  options = {:d => nil}.merge(args.last.is_a?(Hash) ? args.pop : {})
  target = self

  while target && m = args.shift
    target = target.send(m) if target.respond_to?(m)
  end

  return target if target
  return options[:d].call if options[:d].respond_to? :call
  return options[:d]
end
